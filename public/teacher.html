// النطاق الفعلي للمحرك الخلفي على Cloudflare Worker const API_BASE = "https://workers.dev"; // 1️⃣ جدار حماية لوحة الإدارة: فحص صلاحيات الأستاذ والتوكن ومنع دخول الطلاب const token = localStorage.getItem('drasty_token'); const role = localStorage.getItem('drasty_role'); const userName = localStorage.getItem('drasty_name'); if (!token || (role !== 'teacher' && role !== 'admin')) { alert("⚠️ اختراق أمني: غير مسموح لك بدخول لوحة الأستاذ. تم تسجيل هذه المحاولة."); localStorage.clear(); window.location.href = "auth.html"; } // تثبيت اسم الأستاذ أو الموجه في الهيدر الزجاجي العلوي document.getElementById('admin-display-name').innerText = `${userName || 'الموجه المسؤول'} (${role === 'admin' ? 'مدير' : 'أستاذ'})`; // دالة تسجيل الخروج ومسح الجلسة الرقمية لحماية النظام الخلفي للأكاديمية function executeAdminLogout() { localStorage.clear(); window.location.href = "auth.html"; } // التبديل الحركي النظيف بين تبويبات لوحة التحكم function switchTeacherTab(target) { const btnStudents = document.getElementById('tab-students-btn'); const btnCourses = document.getElementById('tab-courses-btn'); const panelStudents = document.getElementById('panel-students'); const panelCourses = document.getElementById('panel-courses'); if (target === 'students') { btnStudents.classList.add('active'); btnCourses.classList.remove('active'); panelStudents.classList.remove('hidden'); panelCourses.classList.add('hidden'); fetchAndRenderStudents(); // تحديث قائمة الطلاب تلقائياً عند فتح التبويب } else { btnStudents.classList.remove('active'); btnCourses.classList.add('active'); panelStudents.classList.add('hidden'); panelCourses.classList.remove('hidden'); } } // 2️⃣ محرك سحب حسابات الطلاب وإعادة بنائها بداخل الجدول الزجاجي async function fetchAndRenderStudents() { const tableBody = document.getElementById('students-table-body'); try { const response = await fetch(`${API_BASE}/api/admin/students`, { headers: { 'Authorization': token } }); if (!response.ok) throw new Error(); const students = await response.json(); if (students.length === 0) { tableBody.innerHTML = `لا يوجد طلبات انضمام أو حسابات طلاب مسجلة حالياً.`; return; } tableBody.innerHTML = ""; students.forEach(student => { const gradeText = student.grade === 'baccalaureate' ? 'بكالوريا' : 'تاسع'; const branchText = student.branch === 'scientific' ? 'علمي' : (student.branch === 'literary' ? 'أدبي' : 'عام'); // حساب الأيام المتبقية ديناميكياً const diffTime = new Date(student.subscription_expires_at) - new Date(); const diffDays = Math.ceil(diffTime / (1000 * 60 * 60 * 24)); const daysRemaining = diffDays > 0 ? `${diffDays} يوم` : "منتهي الصلاحية"; const tr = document.createElement('tr'); tr.innerHTML = ` ${student.name} ${student.phone} ${student.email} ${gradeText} - ${branchText} ${daysRemaining} ${student.status === 'pending' ? `` : `` } `; tableBody.appendChild(tr); }); } catch (err) { tableBody.innerHTML = `فشل جلب الحسابات، تحقق من استجابة السيرفر الخلفي.`; } } // دالة تحديث حالة الطالب الحية (موافقة / حظر) بداخل قاعدة البيانات D1 async function updateStudentStatus(userId, targetStatus) { try { const response = await fetch(`${API_BASE}/api/admin/approve`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': token }, body: JSON.stringify({ userId, targetStatus }) }); if (response.ok) { alert("تم تحديث حالة الطالب بنجاح في قاعدة البيانات الحية."); fetchAndRenderStudents(); } else { alert("فشلت العملية الإدارية."); } } catch (e) { alert("حدث خطأ في الاتصال السحابي."); } } // دالة تمديد أيام صلاحية اشتراك الطالب لـ 30 يوماً إضافية ديناميكياً async function extendStudentDays(userId) { try { const response = await fetch(`${API_BASE}/api/admin/extend`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': token }, body: JSON.stringify({ userId, daysToAdd: 30 }) }); if (response.ok) { alert("تم تمديد صلاحية اشتراك الطالب المختار بـ 30 يوماً بنجاح."); fetchAndRenderStudents(); } else { alert("فشل تمديد الاشتراك."); } } catch (e) { alert("خطأ في الاتصال بالخادم السحابي."); } } // 3️⃣ معالجة طلب حقن ونشر درس جديد في نظام حصص الـ 60 دقيقة async function handleCourseInjection(event) { event.preventDefault(); const subject_name = document.getElementById('course-subject').value; const grade_target = document.getElementById('course-grade').value; const branch_target = document.getElementById('course-branch').value; const title = document.getElementById('course-title').value; const duration_minutes = document.getElementById('course-duration').value; const video_url = document.getElementById('course-video-url').value; try { const response = await fetch(`${API_BASE}/api/courses/inject`, { method: 'POST', headers: { 'Content-Type': 'application/json', 'Authorization': token }, body: JSON.stringify({ subject_name, grade_target, branch_target, title, duration_minutes, video_url }) }); if (response.ok) { alert("🚀 إمبراطوري! تم حقن الحصة بنجاح وتوزيعها على الطلاب المعتمدين في فئتها."); document.getElementById('form-inject-course').reset(); } else { const data = await response.json(); alert(`❌ فشل الحقن: ${data.error}`); } } catch (err) { alert("❌ حدث خطأ، الخادم السحابي غير مستجيب لحقن المناهج."); } } // تشغيل سحب قائمة الطلاب بمجرد تحميل لوحة الأستاذ بنجاح window.onload = function() { fetchAndRenderStudents();
